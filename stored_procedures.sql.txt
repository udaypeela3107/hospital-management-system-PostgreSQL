CREATE OR REPLACE FUNCTION get_patient_appointments(p_id INT)
RETURNS TABLE (
    appointment_id INT,
    appointment_date DATE,
    status VARCHAR
)
AS $$
BEGIN
    RETURN QUERY
    SELECT
        a.appointment_id,
        a.appointment_date,
        a.status
    FROM appointments a
    WHERE a.patient_id = p_id;
END;
$$ LANGUAGE plpgsql;
